Este algoritmo iserta un nodo en unua lista ordenada, mas no desordena la lista
 * creamos i com apuntador  del primer nodo
 * dato representa el contenido del nuevo nodo
 * v, k y l son variables apuntador, las cuales se van a desplazar en la lista.
 * num y liga son campos de los nodo de la lista

1.Hacer band<-1 y v<-i
2.mientras((v->num≠x)&&(band==1)) Repetir
   2.1Si(v^.liga≠NULL)
      entonces
         l<-v
         v<-v^.liga
      else
         band<-0
         break
   2.2 fin del condicional del paso 2.1
3.fin del ciclo del paso 2   
4.Si(band=1)
   entonces
   4.1Si(i=v)
      entonces
         k<-creaNodo()
         k^.num<-dato
         k^.liga<-i
         i<-k
      Si no
         4.1.1.Si((l^.num>dato)&&(v^.num<dato))
         entonces
            k<-creaNodo()
            k^.num<-dato
            l^.liga<-k
            k^.liga<-v
         Sino
            Escribir("El numero que ingreso es invalido")
         4.1.2 fin de la condicional del paso 4.1.1.  
      4.2 fin del condicional 4.1 
   else
      Escribir("El numero dado no se encuentra")
   5. fin del condicional del paso 4.
